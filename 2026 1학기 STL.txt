#include <iostream>
#include <string>
#include <fstream>
#include <random>
#include <print>
#include <array>
#include <algorithm>

#include "save.h"

std::default_random_engine dre;
std::uniform_int_distribution uid{ 0, 9999 };
std::uniform_int_distribution uidNameLen{ 10, 30 };
std::uniform_int_distribution<int> uidChar{ 'a', 'z' };

class Dog {
public:
	Dog( ) {
		num = uid(dre);
		size_t len = uidNameLen(dre);
		for ( int i = 0; i < len; ++i ) {
			name += uidChar(dre);
		}
	}

	std::string getName() const {
		return name;
	}
private:
	std::string name;		// 10 이상 30 이하 임의의 소문자
	int num{ uid(dre) };

	friend std::ostream& operator<<(std::ostream& os , const Dog& dog) {
		std::print(os , "[{:4}] - {}" , dog.num , dog.name);
		return os;
	}
};


int main()
{
	std::array<Dog , 1000> dogs;
	
	std::sort(dogs.begin( ) , dogs.end( ) , [](const Dog& a , const Dog& b) {
		return a.getName( ) < b.getName( );
		});

	for ( const Dog& dog : dogs ) {
		std::cout << dog << std::endl;
	}

	save( "main.cpp" );		// "메인.cpp"를 저장하자
}
